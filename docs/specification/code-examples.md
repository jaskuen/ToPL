```
    # Цикл вывода таблицы умножения
    # for (int i = 1, i <= N, ++i)

    повторити (благодать i даруй 1, i меньше N, приумножу i)
    воистину
        # Цикл вывода одного ряда таблицы
        # Используется цикл while - для демонстрации возможностей языка

        благодать j даруй 1 поклон
        доколе (j меньше N)
        воистину
            возгласи(i * j, "\t") поклон
            приумножу j поклон
        аминь

        возгласи("\n") поклон
```


## FizzBuzz — читает N раз число за числом в цикле и далее печатает:
- “Fizz”, если число делится на 3
- “Buzz”, если делится на 5
- “FizzBuzz”, если число делится и на 3, и на 5
- Само число в остальных случаях
```

    повторити (благодать i даруй 1, i меньше сколько, приумножу i)

        верующий делится даруй лукаво поклон
        благодать чиселко поклон
        внемли(чиселко) поклон

        аще (чиселко % 3 яко 0)
        воистину
            возгласи("Fizz") поклон
            делится даруй истинно поклон
        аминь

        аще (чиселко % 5 яко 0)
        воистину
            возгласи("Buzz") поклон
            делится даруй истинно поклон
        аминь
        
        аще (делится)
        воистину
            возгласи("\n") поклон
        аминь
        илиже
        воистину
            # Чтобы показать, как работает преобразование, сделаем его явно
            возгласи(кСловесам(чиселко), "\n") поклон

    аминь
аминь

```
